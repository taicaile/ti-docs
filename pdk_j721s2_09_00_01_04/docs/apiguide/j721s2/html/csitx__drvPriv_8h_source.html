72 /*
73  * State transitions are explained below
74  *
75  * IDLE - initial or not open state
76  * |
77  * CREATED - state after calling Fvid2_create
78  * |
79  * DO_START - temporary state when Fvid2_start() is called
80  * | this means driver has requested Transmit Module to start the instance
81  * |
82  * RUNNING - this is the state after Fvid2_start() completes, this means
83  * | Transmit Module started capture and now it is running, it remains in this
84  * | state
85  * | until Fvid2_stop() is called
86  * |
87  * DO_STOP - temporary state when Fvid2_stop is called()
88  * | this means driver has request Transmit Module to stop the instance
89  * |
90  * STOPPED - this is the state after Fvid2_stop() completes, this means
91  * | Transmit Module has stopped this instance
92  * |
93  * IDLE - this is the state after Fvid2_delete() completes
94  */

283 typedef struct

285  /*
286  * Initially all the queue elements will be in freeQ.
287  *
288  * If serialized multi-channel transitions is disabled
289  * For every application queue operation and
290  * For every request submitted to core, freeQ -> curQ
291  * For every request completion from core, curQ -> doneQ
292  * For every application dequeue operation, doneQ -> freeQ
293  *
294  * If serialized multi-channel transitions is enabled
295  * For every application queue operation and
296  * For every request submitted to core, freeQ -> waitQ, waitQ -> curQ
297  * (in case of first frame/pipe up stage)
298  * For every request completion from core, curQ -> doneQ, waitQ -> curQ
299  * For every application dequeue operation, doneQ -> freeQ
300  */

303  Fvid2UtilsLinkListObj freeLlObj;
