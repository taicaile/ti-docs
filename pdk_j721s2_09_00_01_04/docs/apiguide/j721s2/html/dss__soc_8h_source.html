472 static inline void Dss_irqParamsInit(Dss_IrqParams *irqParams)
473 {
474  if(NULL != irqParams)
475  {
476  irqParams->dssCommonRegionId = CSL_DSS_COMM_REG_ID_0;
477  irqParams->numValidIrq = DSS_EVT_MGR_INST_ID_MAX;
478  irqParams->irqNum[DSS_EVT_MGR_INST_ID_FUNC] = DSS_FUNC_IRQ_DEFAULT_NUM;
479  irqParams->irqNum[DSS_EVT_MGR_INST_ID_SAFETY] =
480  DSS_SAFE_IRQ_DEFAULT_NUM;

482  DSS_SECURE_IRQ_DEFAULT_NUM;
483  }
484 }

486 static inline void Dss_rmInfoInit(Dss_RmInfo *rmInfo)
487 {
488  uint32_t i = 0U;
489  if(NULL != rmInfo)
490  {
491  for(i=CSL_DSS_COMM_REG_ID_0; i<CSL_DSS_COMM_REG_ID_MAX; i++)
492  {
493  rmInfo->isCommRegAvailable[i] = TRUE;
494  }
495  for(i=CSL_DSS_VID_PIPE_ID_VID1; i<CSL_DSS_VID_PIPE_ID_MAX; i++)
496  {
497  rmInfo->isPipeAvailable[i] = TRUE;
498  }
499  for(i=CSL_DSS_OVERLAY_ID_1; i<CSL_DSS_OVERLAY_ID_MAX; i++)
500  {
501  rmInfo->isOverlayAvailable[i] = TRUE;
502  }
503  for(i=CSL_DSS_VP_ID_1; i<CSL_DSS_VP_ID_MAX; i++)
504  {
505  rmInfo->isPortAvailable[i] = TRUE;
506  }
507  for(i=CSL_DSS_WB_PIPE_ID_1; i<CSL_DSS_WB_PIPE_ID_MAX; i++)
508  {
509  rmInfo->isWbPipeAvailable[i] = TRUE;
510  }
511  }
512 }
