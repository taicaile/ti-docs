482 static inline void CSL_dssWbPipeCfgInit(
483  CSL_DssWbPipeCfg *wbCfg)
484 {
485  if(NULL != wbCfg)
486  {
487  Fvid2Format_init(&wbCfg->inFmt);
488  Fvid2Format_init(&wbCfg->outFmt);
489  Fvid2PosConfig_init(&wbCfg->inPos);
490  wbCfg->wbMode = CSL_DSS_WB_PIPE_MODE_CAPTURE;
491  wbCfg->yuvAlign = CSL_DSS_WB_PIPE_YUV_ALIGN_LSB;
492  wbCfg->cscRange = CSL_DSS_CSC_RANGE_FULL;
493  wbCfg->scEnable = FALSE;
494  wbCfg->alphaEnable = FALSE;
495  }
496 }
