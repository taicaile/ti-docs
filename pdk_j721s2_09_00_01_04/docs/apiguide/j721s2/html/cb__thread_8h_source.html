195 static inline bool cb_waitpoint_check(cb_waitpoint_t *wp, uint64_t time){
196  if(wp->wakeup==true){
197  // check time diff, consider wrap around to uint64_t max value
198  // unsigned substract typecasted to unsigned stores the 2s complement
199  if(((uint64_t)(time-wp->time)<((uint64_t)1<<63))){
200  wp->wakeup=false;
201  return true;
202  }
203  }
204  return false;
205 }
