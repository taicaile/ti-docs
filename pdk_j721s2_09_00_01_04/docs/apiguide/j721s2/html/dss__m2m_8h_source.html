347 static inline void Dss_m2mCreateParamsInit(Dss_WbCreateParams *createParams)
348 {
349  uint32_t loopCnt = 0U;
350 
351  if(NULL != createParams)
352  {
353  createParams->numPipe = 1U;
354  for (loopCnt = 0U ;
355  loopCnt < DSSM2M_NUM_PIPELINE_TO_USE_IN_M2M_MODE ;
356  loopCnt++)
357  {
358  createParams->pipeId[loopCnt] = CSL_DSS_VID_PIPE_ID_MAX;
359  }
360  createParams->overlayId = CSL_DSS_OVERLAY_ID_1;
361  }
362 }
363 
364 static inline void Dss_m2mPipeCfgParamsInit(Dss_WbPipeCfgParams *cfgParams)
365 {
366  if(NULL != cfgParams)
367  {
368  CSL_dssWbPipeCfgInit(&(cfgParams->pipeCfg));
369  /* dataFormat is fixed as Display pipe/overlay can only output in
370  ARGB48 format */
371  cfgParams->pipeCfg.inFmt.dataFormat = FVID2_DF_ARGB48_12121212;
372  /* Set mode to M2M mode */
373  cfgParams->pipeCfg.wbMode = CSL_DSS_WB_PIPE_MODE_M2M;
374  }
375 }
