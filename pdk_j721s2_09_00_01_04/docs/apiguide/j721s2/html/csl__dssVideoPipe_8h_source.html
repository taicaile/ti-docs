615 static inline void CSL_dssVidPipeCfgInit(
616  CSL_DssVidPipeCfg *pipeCfg)
617 {
618  uint32_t i;
619  if(NULL != pipeCfg)
620  {
621  Fvid2Format_init(&pipeCfg->inFmt);
622  pipeCfg->pipeType = CSL_DSS_VID_PIPE_TYPE_VID;
623  pipeCfg->pixelInc = 1U;
624  pipeCfg->yuvAlign = CSL_DSS_VID_PIPE_YUV_ALIGN_LSB;
625  pipeCfg->outWidth = 800U;
626  pipeCfg->outHeight = 480U;
627  pipeCfg->scEnable = CSL_DSS_VID_PIPE_SC_DISABLED;
628  pipeCfg->cscRange = CSL_DSS_CSC_RANGE_FULL;
629  CSL_dssCscCoeffInit(&pipeCfg->custCscCoeff);
630  pipeCfg->custCscCoeff.cscRange = CSL_DSS_CSC_RANGE_CUSTOM;
631  pipeCfg->flipType = FVID2_FLIP_TYPE_NONE;
632  for(i=0U; i<CSL_DSS_NUM_LUT_ENTRIES; i++)
633  {
634  pipeCfg->clutData[i] = 0x0U;
635  }
636  pipeCfg->nibbleModeEnable = FALSE;
637  pipeCfg->gammaEnable = FALSE;
638  }
639 }
640 
641 static inline void CSL_dssVidPipeDmaCfgInit(
642  CSL_DssVidPipeDmaCfg *dmaCfg)
643 {
644  if(NULL != dmaCfg)
645  {
646  dmaCfg->bufPreloadControl = CSL_DSS_VID_PIPE_PRELOAD_CONTROL_HW;
647  dmaCfg->bufLowThreshold = 0x09FFU;
648  dmaCfg->bufHighThreshold = 0x09F8U;
649  dmaCfg->selfRefreshControl = CSL_DSS_VID_PIPE_REFRESH_CONTROL_SW;
650  dmaCfg->selfRefreshEnable = FALSE;
651  dmaCfg->arbitration = CSL_DSS_VID_PIPE_PRIORITY_NORMAL;
652  }
653 }
