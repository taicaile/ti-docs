530 static inline void CSL_druChSubmitTr(const CSL_DRU_t *pRegs,
531  uint32_t chId,
532  uint32_t coreId,
533  const CSL_UdmapTR *tr)
534 {
535  const CSL_DRU_CHCORERegs_CHCORE_CORE *pCoreRegs =
536  &pRegs->CHCORE[chId].CORE[coreId];
537  const uint32_t *trWord32 = (const uint32_t *) tr;
538  volatile uint32_t *submitWord32 = (volatile uint32_t *) pCoreRegs;
539  uint32_t numWords32, cnt;
540 
541  /* Write all wrds expect first */
542  numWords32 = (sizeof(CSL_UdmapTR) / sizeof(uint32_t));
543  for(cnt = 1U; cnt < numWords32; cnt++)
544  {
545  CSL_REG32_WR(submitWord32+cnt, trWord32[cnt]);
546  }
547  /* This triggers the actual TR submission */
548  CSL_REG32_WR(submitWord32, trWord32[0]);
549 
550  return;
551 }
