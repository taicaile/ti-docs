79 static inline float my_OneByX(float x)
80 {
81 #ifdef DSP_INTRINSIC_FOR_RECIPORCAL
82  float TWO = 2.0f;
83  float X;
84 
85 #ifdef RCP_BOUND_CHK
86  float Big = FLT_MAX;
87  float Y;
88  Y = _fabsf(x);
89 #endif
90 
91  X = __recip(x);
92  X = X * (TWO - x*X);
93  X = X * (TWO - x*X);
94 
95 #ifdef RCP_BOUND_CHK
96  if (Y > Big) {
97  X = 0.0f;
98  }
99  if((x == 0.0f) | (x == -0.0f))
100  X = Big;
101 #endif
102 
103  return (X);
104 #else
105  float den;
106  den = 1/x;
107  return(den);
108 #endif
109 }
