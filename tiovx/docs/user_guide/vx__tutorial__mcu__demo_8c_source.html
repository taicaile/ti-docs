72 #if defined(SOC_AM62A)
73 void vx_tutorial_mcu_demo(vx_bool use_mcu1_core)
74 #else
75 void vx_tutorial_mcu_demo(vx_bool use_mcu3_core)
76 #endif
77 {
78  vx_graph graph;
79  vx_context context;
80  vx_uint8 scalar_val = 0;
81  vx_scalar scalar, scalar_out;
82  vx_uint8 output = 0;
83  vx_uint32 num_streams = 0, golden_output = 0;
84  vx_node node0, node1;
85  vx_status status;
86 
87  printf(" vx_tutorial_mcu_demo: Tutorial Started !!! \n");
95  context = vxCreateContext();
98  tivxTestKernelsLoadKernels(context);
99 
105  graph = vxCreateGraph(context);
108  scalar = vxCreateScalar(context, VX_TYPE_UINT8, &scalar_val);
109  scalar_out = vxCreateScalar(context, VX_TYPE_UINT8, &scalar_val);
110 
116  node0 = tivxScalarSourceNode(graph, scalar);
117  vxSetReferenceName((vx_reference)node0, "Scalar_Source 1");
118  #if defined(SOC_AM62A)
119  if (vx_true_e == use_mcu1_core)
120  {
121  vxSetNodeTarget(node0, VX_TARGET_STRING, TIVX_TARGET_MCU1_0);
122  }
123  #else
124  if (vx_true_e == use_mcu3_core)
125  {
126  if(tivxIsTargetEnabled(TIVX_TARGET_MCU3_0))
127  {
128  vxSetNodeTarget(node0, VX_TARGET_STRING, TIVX_TARGET_MCU3_0);
129  }
130  else
131  {
132  vxSetNodeTarget(node0, VX_TARGET_STRING, TIVX_TARGET_MCU3_1);
133  }
134  }
135  #endif
136  #if defined(SOC_J784S4)
137  else
138  {
139  if(tivxIsTargetEnabled(TIVX_TARGET_MCU4_0))
140  {
141  vxSetNodeTarget(node0, VX_TARGET_STRING, TIVX_TARGET_MCU4_0);
142  }
143  else
144  {
145  vxSetNodeTarget(node0, VX_TARGET_STRING, TIVX_TARGET_MCU4_1);
146  }
147  }
148  #endif
149 
150  node1 = tivxScalarIntermediateNode(graph, scalar, scalar_out);
151  vxSetReferenceName((vx_reference)node1, "Scalar_Source 2");
152  #if defined(SOC_AM62A)
153  if (vx_true_e == use_mcu1_core)
154  {
155  vxSetNodeTarget(node1, VX_TARGET_STRING, TIVX_TARGET_MCU1_0);
156  }
157  #else
158  if (vx_true_e == use_mcu3_core)
159  {
160  if(tivxIsTargetEnabled(TIVX_TARGET_MCU3_0))
161  {
162  vxSetNodeTarget(node1, VX_TARGET_STRING, TIVX_TARGET_MCU3_0);
163  }
164  else
165  {
166  vxSetNodeTarget(node1, VX_TARGET_STRING, TIVX_TARGET_MCU3_1);
167  }
168  }
169  #endif
170  #if defined(SOC_J784S4)
171  else
172  {
173  if(tivxIsTargetEnabled(TIVX_TARGET_MCU4_0))
174  {
175  vxSetNodeTarget(node1, VX_TARGET_STRING, TIVX_TARGET_MCU4_0);
176  }
177  else
178  {
179  vxSetNodeTarget(node1, VX_TARGET_STRING, TIVX_TARGET_MCU4_1);
180  }
181  }
182  #endif
183 
186  status = vxEnableGraphStreaming(graph, node0);
194  status = vxVerifyGraph(graph);
204  show_graph_attributes(graph);
214  show_node_attributes(node0);
215  show_node_attributes(node1);
216 
217  printf("Input scalar value = %d\n", scalar_val);
218 
219  if(status==(vx_status)VX_SUCCESS)
220  {
221  printf(" Executing graph ...\n");
222 
223  vxStartGraphStreaming(graph);
224 
225  tivxTaskWaitMsecs(50);
226 
227  vxStopGraphStreaming(graph);
228 
229  printf(" Executing graph ... Done !!!\n");
230 
238  show_graph_attributes(graph);
248  show_node_attributes(node0);
249  show_node_attributes(node1);
252  vxQueryGraph(graph, TIVX_GRAPH_STREAM_EXECUTIONS, &num_streams, sizeof(num_streams));
253 
254  if (num_streams != 0)
255  {
256  golden_output = num_streams % 256;
257  vxCopyScalar(scalar_out, &output, (vx_enum)VX_READ_ONLY, (vx_enum)VX_MEMORY_TYPE_HOST);
258  printf("Output scalar value = %d\n", output);
259  if(output!=golden_output)
260  {
261  printf("ERROR: %d != %d\n", output, golden_output);
262  }
263  else
264  {
265  printf("RUN SUCCESS!!!\n");
266  }
267  }
268  }
269 
276  vxReleaseScalar(&scalar);
277  vxReleaseScalar(&scalar_out);
278  vxReleaseNode(&node0);
279  vxReleaseNode(&node1);
280 
287  vxReleaseGraph(&graph);
290  tivxTestKernelsUnLoadKernels(context);
291 
300  vxReleaseContext(&context);
303  printf(" vx_tutorial_mcu_demo: Tutorial Done !!! \n");
304  printf(" \n");
305 }
