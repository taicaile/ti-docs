158  {
159  size = 2;
160  }
161  else if(type == (uint32_t)DSPLIB_UINT24)
162  {
163  size = 3;
164  }
165  else if((type == (uint32_t)DSPLIB_INT32) ||
166  (type == (uint32_t)DSPLIB_UINT32) ||
167  (type == (uint32_t)DSPLIB_FLOAT32))
168  {
169  size = 4;
170  }
171  else if((type == (uint32_t)DSPLIB_INT64) ||
172  (type == (uint32_t)DSPLIB_UINT64) ||
173  (type == (uint32_t)DSPLIB_FLOAT64))
174  {
175  size = 8;
176  }
177  else if((type == (uint32_t)DSPLIB_INT128) ||
178  (type == (uint32_t)DSPLIB_UINT128))
179  {
180  size = 16;
181  }
182  else {
183  size = 1;
184  }
